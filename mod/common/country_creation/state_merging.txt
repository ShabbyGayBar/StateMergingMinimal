# ---- 00_releasable_countries.txt ----
#   Root                     = the country that would give up the land
#
#   scope:war_goal_holder    = The country enforcing/pressing a war goal.
#                              this scope is not set when the release is done outside of a war goal context.
# 
# Can have assymetric requirements for self releasing vs being forced to release, otherwise do as follows.
# e.g.:
#    OR = {
#			scope:war_goal_holder ?= {
#				has_technology_researched = pan-nationalism
#			}
#			AND = {
#				NOT = { exists = scope:war_goal_holder }
#				has_technology_researched = pan-nationalism
#			}
#		}

REPLACE:UBD = {
	states = { STATE_KAUNAS STATE_VILNIUS STATE_TALINN STATE_TARTU STATE_RIGA STATE_COURLAND }

	required_num_states = 3

	ai_will_do = { always = no }

	possible = {
		# Checks if the technology is researched by the war goal holder (if any) or by the releasing country
		OR = {
			scope:war_goal_holder ?= {
				has_technology_researched = pan-nationalism
			}
			AND = {
				NOT = { exists = scope:war_goal_holder }
				has_technology_researched = pan-nationalism
			}
		}
	}
}
